1"""Transient email verification service. 

2 

3Email addresses are NEVER stored — they are held in memory only for the 

4duration of the HTTP request that sends the verification email. 

5""" 

6 

7import logging 

8 

9from django.conf import settings 

10from django.core.mail import send_mail 

11from django.core.signing import TimestampSigner, BadSignature, SignatureExpired 

12from django.template.loader import render_to_string 

13 

14logger = logging.getLogger(__name__) 

15 

16VERIFICATION_MAX_AGE = 7200 # 2 hours in seconds 

17 

18_signer = TimestampSigner(salt="email-verification") 

19 

20 

21def generate_verification_token(user_id: int) -> str: 

22 """Generate a signed, timestamped token encoding the user's PK.""" 

23 return _signer.sign(str(user_id)) 

24 

25 

26def validate_verification_token(token: str) -> int | None: 

27 """Validate and unsign a verification token. 

28 

29 Returns the user_id if valid, None if expired/tampered. 

30 """ 

31 try: 

32 value = _signer.unsign(token, max_age=VERIFICATION_MAX_AGE) 

33 return int(value) 

34 except (BadSignature, SignatureExpired, ValueError): 

35 return None 

36 

37 

38def send_verification_email(user_id: int, email: str) -> None: 

39 """Send a verification email. The email parameter is transient — not stored. 

40 

41 Args: 

42 user_id: The user's primary key (encoded in the token). 

43 email: The recipient email address (used only for sending, then discarded). 

44 """ 

45 token = generate_verification_token(user_id) 

46 verify_url = f"{settings.SITE_URL}/api/auth/verify-email/?token={token}" 

47 

48 context = {"verify_url": verify_url, "expiry_hours": 2} 

49 

50 text_body = render_to_string("core/verification_email.txt", context) 

51 html_body = render_to_string("core/verification_email.html", context) 

52 

53 send_mail( 

54 subject="Verify your Cookie account", 

55 message=text_body, 

56 from_email=settings.DEFAULT_FROM_EMAIL, 

57 recipient_list=[email], 

58 html_message=html_body, 

59 fail_silently=False, 

60 ) 

61 # email parameter goes out of scope here — never stored 

62 logger.info("Verification email sent for user_id=%s", user_id)

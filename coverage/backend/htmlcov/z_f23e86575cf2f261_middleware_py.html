9def get_client_ip(request): 

10 """Extract the real client IP from X-Forwarded-For for django-ratelimit. 

11 

12 X-Forwarded-For may contain multiple IPs: "client, proxy1, proxy2". 

13 We take the leftmost entry (the original client), validate it as a 

14 real IP address, and return it. Falls back to REMOTE_ADDR if the 

15 header is missing or every entry is malformed. 

16 

17 Used via RATELIMIT_IP_META_KEY = "apps.core.middleware.get_client_ip" 

18 """ 

19 forwarded_for = request.META.get("HTTP_X_FORWARDED_FOR", "") 

20 if forwarded_for: 

21 # Take the leftmost (client) IP, strip whitespace 

22 candidate = forwarded_for.split(",")[0].strip() 

23 try: 

24 ipaddress.ip_address(candidate) 

25 return candidate 

26 except ValueError: 

27 pass 

28 return request.META.get("REMOTE_ADDR", "127.0.0.1")

44class DeviceDetectionMiddleware: 

45 """Middleware to detect legacy browsers that can't run modern React. 

46 

47 Sets request.is_legacy_device = True for browsers that need the legacy frontend: 

48 - iOS < 11 (Safari lacks ES6 module support) 

49 - Internet Explorer (all versions) 

50 - Edge Legacy (non-Chromium, pre-2020) 

51 - Chrome < 60 

52 - Firefox < 55 

53 

54 Note: Actual redirects are handled by Nginx for performance (see nginx/nginx.conf). 

55 This middleware provides the detection flag for use in views/templates if needed. 

56 """

74 def _is_legacy_device(self, request): 

75 """Check if the request is from a browser that can't run modern React.""" 

76 user_agent = request.META.get("HTTP_USER_AGENT", "") 

77 if not user_agent: 

78 return False

80 detectors = [ 

81 self._is_legacy_ios, 

82 self._is_internet_explorer, 

83 self._is_edge_legacy, 

84 self._is_old_chrome, 

85 self._is_old_firefox,

91 match = self.IOS_PATTERN.search(ua)

95 """Internet Explorer (all versions)."""

97 

98 def _is_edge_legacy(self, ua): 

99 """Edge Legacy (non-Chromium, pre-2020).""" 

100 return "Edge/" in ua and "Edg/" not in ua

103 """Chrome < 60 (excluding Edge and Opera).""" 

104 if "Chrome/" not in ua or "Edg" in ua or "OPR/" in ua:

115def _ninja_path_to_regex(path: str) -> re.Pattern: 

116 """Convert a Ninja URL path (with `{param}` / `{param:int}` placeholders) 

117 to a compiled regex that matches a single URL, with an optional trailing 

118 slash. A `{param}` spans exactly one path segment (no slashes).""" 

119 pattern = re.sub(r"\{[^/}]+\}", r"[^/]+", path.rstrip("/")) 

120 return re.compile(rf"^{pattern}/?$")

126def _home_only_patterns() -> tuple[re.Pattern, ...]: 

127 """Introspect the Ninja API once to compile regexes for every path 

128 whose registered methods ALL use `HomeOnlyAuth`. A path that mixes 

129 `HomeOnlyAuth` methods with `SessionAuth` methods (e.g. `/api/ai/quotas` 

130 has GET=SessionAuth + PUT=HomeOnlyAuth) is NOT included here — those 

131 paths must remain reachable in passkey mode for the non-gated methods. 

132 Lazy-imported to avoid a circular import at module load (urls → ninja 

133 routers → auth classes → middleware).""" 

134 global _home_only_patterns_cache 

135 if _home_only_patterns_cache is not None: 

136 return _home_only_patterns_cache 

137 

138 from apps.core.auth import HomeOnlyAuth 

139 from cookie.urls import api 

140 

141 paths: set[str] = set() 

142 for prefix, router in api._routers: 

143 for path, path_op in router.path_operations.items(): 

144 if not path_op.operations: 

145 continue 

146 all_home_only = all( 

147 any(isinstance(a, HomeOnlyAuth) for a in (op.auth_callbacks or [])) for op in path_op.operations 

148 ) 

149 if all_home_only: 

150 paths.add(f"/api{prefix}{path}") 

151 

152 _home_only_patterns_cache = tuple(_ninja_path_to_regex(p) for p in paths) 

153 return _home_only_patterns_cache

156class HomeOnlyRouteGateMiddleware: 

157 """Short-circuit every method on HomeOnlyAuth-gated routes to 404 in 

158 non-home auth modes, above Django's URL dispatcher. 

159 

160 Without this, HEAD/OPTIONS (and other unregistered methods) on gated 

161 routes fall through to Django's 405 handler, which emits an `Allow:` 

162 header listing the real method set — leaking route existence to 

163 unauthenticated probes. The v1.42.0 security posture promises that 

164 gated routes are "indistinguishable from never-existed paths"; this 

165 middleware upholds that promise for every verb, not just those with 

166 a registered Ninja handler. 

167 """ 

168 

169 def __init__(self, get_response): 

170 self.get_response = get_response 

171 

172 def __call__(self, request): 

173 if settings.AUTH_MODE != "home": 

174 path = request.path 

175 for pattern in _home_only_patterns(): 

176 if pattern.match(path): 

177 return JsonResponse({"detail": "Not found"}, status=404) 

178 return self.get_response(request)

181class MethodNotAllowedToNotFoundMiddleware: 

182 """Rewrite every HTTP 405 response to a JSON 404. 

183 

184 Rationale: a 405 "Method Not Allowed" response tells a probe that the 

185 URL exists but doesn't accept this method. For POST-only endpoints 

186 like `/api/auth/logout/` and `/api/auth/device/authorize/`, an 

187 unauthenticated `GET` would otherwise return 405 — contradicting the 

188 v1.42.0 "gated paths are indistinguishable from never-existed paths" 

189 invariant that `HomeOnlyRouteGateMiddleware` established for 

190 HomeOnlyAuth routes (pentest round 6 / F-5). 

191 

192 The rewrite is global rather than per-endpoint because: 

193 - Cookie's clients (React frontend, legacy ES5 frontend) never rely 

194 on 405 for control flow — they only issue documented methods. 

195 - A global rule removes the possibility of forgetting to gate a new 

196 POST-only endpoint. 

197 - The existing `Allow:` header that Django attaches to 405 responses 

198 also leaks the method set; collapsing to 404 drops it. 

199 

200 Nginx-generated 405s (none occur in the current config, but a future 

201 location block could introduce one) are separately rewritten to 404 

202 via `error_page 405 =404 @not_found;` in `nginx/nginx.prod.conf`. 

203 """ 

204 

205 def __init__(self, get_response): 

206 self.get_response = get_response 

207 

208 def __call__(self, request): 

209 response = self.get_response(request) 

210 if response.status_code == 405: 

211 return JsonResponse({"detail": "Not found"}, status=404) 

212 return response
